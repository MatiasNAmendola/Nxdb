00031         {
00032             if (target == null) return;
00033 
00034             // Get all values first so if something goes wrong we haven't started modifying the object
00035             List<KeyValuePair<MemberInfo, object>> values
00036                 = new List<KeyValuePair<MemberInfo, object>>();
00037             foreach (KeyValuePair<MemberInfo, PersistentMemberAttribute> kvp
00038                 in typeCache.PersistentMembers.Where(kvp => kvp.Value.Fetch))
00039             {
00040                 Type memberType;
00041                 object member = GetValue(kvp.Key, target, out memberType);
00042                 TypeCache memberTypeCache = memberType != null ? cache.GetTypeCache(memberType) : null;
00043                 object value = kvp.Value.FetchValue(element, member, memberTypeCache, cache);
00044                 if(kvp.Value.Required && value == null) throw new Exception("Could not get required member.");
00045                 values.Add(new KeyValuePair<MemberInfo, object>(kvp.Key, value));
00046             }

00049             foreach (KeyValuePair<MemberInfo, object> value in values)
00050             {
00051                 SetValue(value.Key, target, value.Value);
00052             }
00053         }
